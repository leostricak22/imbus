dockerfile
FROM maven:3.8.3-openjdk-17 AS build
COPY backend/src /home/app/src
COPY backend/pom.xml /home/app
RUN mvn -f /home/app/pom.xml clean package

FROM openjdk:17-jdk-slim
COPY --from=build /home/app/target/imbus.jar /usr/local/lib/imbus.jar
COPY wait-for-it.sh /usr/local/bin/wait-for-it.sh
RUN chmod +x /usr/local/bin/wait-for-it.sh
EXPOSE 8080
ENTRYPOINT ["sh", "-c", "/usr/local/bin/wait-for-it.sh mysqldb:3306 --timeout=30 --strict -- java -jar /usr/local/lib/imbus.jar"]
